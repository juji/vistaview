Learn about all the configuration options available in VistaView.

## Basic Options

### `elements` (required)

Specify which elements to include in the gallery.

```js
// CSS selector
vistaView({
  elements: '#gallery a',
});

// Array of image configurations
vistaView({
  elements: [
    { src: '/photo1.jpg', alt: 'Photo 1' },
    { src: '/photo2.jpg', alt: 'Photo 2' },
  ],
});
```

### `animationDurationBase`

Base duration for animations in milliseconds.

- **Type**: `number`
- **Default**: `333`

```js
vistaView({
  elements: '#gallery a',
  animationDurationBase: 500, // Slower animations
});
```

### `maxZoomLevel`

Maximum zoom multiplier for images.

- **Type**: `number`
- **Default**: `2`

```js
vistaView({
  elements: '#gallery a',
  maxZoomLevel: 3, // Allow 3x zoom
});
```

### `preloads`

Number of adjacent images to preload on each side.

- **Type**: `number`
- **Default**: `1`

```js
vistaView({
  elements: '#gallery a',
  preloads: 2, // Preload 2 images before and after
});
```

### `keyboardListeners`

Enable or disable keyboard navigation.

- **Type**: `boolean`
- **Default**: `true`

```js
vistaView({
  elements: '#gallery a',
  keyboardListeners: false, // Disable keyboard controls
});
```

### `arrowOnSmallScreens`

Show previous/next arrows on screens smaller than 768px.

- **Type**: `boolean`
- **Default**: `false`

```js
vistaView({
  elements: '#gallery a',
  arrowOnSmallScreens: true, // Show arrows on mobile
});
```

### `rapidLimit`

Minimum time in milliseconds between rapid navigation actions.

- **Type**: `number`
- **Default**: `222`

```js
vistaView({
  elements: '#gallery a',
  rapidLimit: 300, // Slow down rapid navigation
});
```

### `initialZIndex`

Starting z-index for the lightbox overlay.

- **Type**: `number`
- **Default**: `undefined`

```js
vistaView({
  elements: '#gallery a',
  initialZIndex: 9999, // Place above other elements
});
```

## Control Configuration

### `controls`

Configure which controls appear in different positions.

- **Type**: `object`
- **Default**: 
  ```js
  {
    topLeft: ['indexDisplay'],
    topRight: ['zoomIn', 'zoomOut', 'close'],
    bottomLeft: ['description'],
  }
  ```

```js
vistaView({
  elements: '#gallery a',
  controls: {
    topLeft: ['indexDisplay'],
    topRight: ['zoomIn', 'zoomOut', 'close'],
    topCenter: [],
    bottomLeft: ['description'],
    bottomCenter: [],
    bottomRight: [],
  },
});
```

Available built-in controls:
- `indexDisplay` - Shows "1 / 5" counter
- `zoomIn` - Zoom in button
- `zoomOut` - Zoom out button
- `close` - Close button
- `description` - Shows image alt text

You can also add extension controls (like `download`).

## Extensions

### `extensions`

Array of extension instances to add functionality.

- **Type**: `VistaExtension[]`
- **Default**: `[]`

```js
import { download } from 'vistaview/extensions/download';
import { youtubeVideo } from 'vistaview/extensions/youtube-video';

vistaView({
  elements: '#gallery a',
  extensions: [download(), youtubeVideo()],
  controls: {
    topRight: ['download', 'zoomIn', 'zoomOut', 'close'],
  },
});
```

## Event Handlers

### `onOpen`

Called when the lightbox opens.

- **Type**: `(vistaView: VistaView) => void`

```js
vistaView({
  elements: '#gallery a',
  onOpen: (vistaView) => {
    console.log('Gallery opened');
    console.log('Current index:', vistaView.getCurrentIndex());
  },
});
```

### `onClose`

Called when the lightbox closes.

- **Type**: `(vistaView: VistaView) => void`

```js
vistaView({
  elements: '#gallery a',
  onClose: (vistaView) => {
    console.log('Gallery closed');
  },
});
```

### `onImageView`

Called when viewing an image (including on initial open).

- **Type**: `(data: VistaData, vistaView: VistaView) => void`

```js
vistaView({
  elements: '#gallery a',
  onImageView: (data, vistaView) => {
    console.log(`Viewing image ${data.index.to + 1}`);
    console.log('Previous index:', data.index.from);
    console.log('Navigation via:', data.via);
  },
});
```

## Advanced Functions

:::caution
These functions allow deep customization but require understanding VistaView internals.
:::

### `initFunction`

Custom initialization logic that runs when the VistaView instance is created.

- **Type**: `(vistaView: VistaView) => void`

### `imageSetupFunction`

Custom image setup logic when navigating between images.

- **Type**: `(data: VistaData) => void`

### `transitionFunction`

Custom transition animation between images.

- **Type**: `(data: VistaData, abortSignal: AbortSignal) => { cleanup: () => void, transitionEnded: Promise<void> } | void`

### `openFunction`

Custom logic when the lightbox opens.

- **Type**: `(vistaView: VistaView) => void`

### `closeFunction`

Custom cleanup logic when the lightbox closes.

- **Type**: `(vistaView: VistaView) => void`

## Complete Example

```js
import { vistaView } from 'vistaview';
import { download } from 'vistaview/extensions/download';
import 'vistaview/style.css';

const gallery = vistaView({
  // Required
  elements: '#gallery a',
  
  // Optional configuration
  animationDurationBase: 400,
  maxZoomLevel: 3,
  preloads: 2,
  keyboardListeners: true,
  arrowOnSmallScreens: true,
  rapidLimit: 250,
  initialZIndex: 9999,
  
  // Controls
  controls: {
    topLeft: ['indexDisplay'],
    topRight: ['download', 'zoomIn', 'zoomOut', 'close'],
    bottomLeft: ['description'],
  },
  
  // Extensions
  extensions: [download()],
  
  // Event handlers
  onOpen: (vistaView) => {
    console.log('Opened at index:', vistaView.getCurrentIndex());
  },
  onClose: (vistaView) => {
    console.log('Gallery closed');
  },
  onImageView: (data, vistaView) => {
    console.log(`Now viewing image ${data.index.to + 1}`);
  },
});

// Programmatic control
gallery.open(0);      // Open at first image
gallery.next();       // Go to next
gallery.prev();       // Go to previous
gallery.zoomIn();     // Zoom in
gallery.zoomOut();    // Zoom out
gallery.close();      // Close
gallery.destroy();    // Clean up
```
