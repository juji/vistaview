## Basic Usage

### With Module Bundler (Vite, Webpack, etc.)

```html
<!-- index.html -->
<div id="gallery">
  <a href="/images/photo1-full.jpg">
    <img src="/images/photo1-thumb.jpg" alt="Photo 1" />
  </a>
  <a href="/images/photo2-full.jpg">
    <img src="/images/photo2-thumb.jpg" alt="Photo 2" />
  </a>
  <a href="/images/photo3-full.jpg">
    <img src="/images/photo3-thumb.jpg" alt="Photo 3" />
  </a>
</div>
```

```js
// main.js
import { vistaView } from 'vistaview';
import 'vistaview/style.css';

vistaView({
  elements: '#gallery a',
});
```

### With CDN (No Build Tools)

```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>VistaView Gallery</title>
  <link rel="stylesheet" href="https://unpkg.com/vistaview/dist/style.css" />
</head>
<body>
  <div id="gallery">
    <a href="/images/photo1-full.jpg">
      <img src="/images/photo1-thumb.jpg" alt="Photo 1" />
    </a>
    <a href="/images/photo2-full.jpg">
      <img src="/images/photo2-thumb.jpg" alt="Photo 2" />
    </a>
    <a href="/images/photo3-full.jpg">
      <img src="/images/photo3-thumb.jpg" alt="Photo 3" />
    </a>
  </div>

  <script src="https://unpkg.com/vistaview/dist/vistaview.umd.js"></script>
  <script>
    // VistaView is available globally
    VistaView.vistaView({
      elements: '#gallery a',
    });
  </script>
</body>
</html>
```

## Advanced Examples

### Programmatic Control

```js
import { vistaView } from 'vistaview';
import 'vistaview/style.css';

const gallery = vistaView({
  elements: '#gallery a',
});

// Open at specific index
document.getElementById('openBtn').addEventListener('click', () => {
  gallery.open(0);
});

// Navigate programmatically
document.getElementById('nextBtn').addEventListener('click', () => {
  gallery.next();
});

document.getElementById('prevBtn').addEventListener('click', () => {
  gallery.prev();
});
```

### Using Image Array

```js
import { vistaView } from 'vistaview';
import 'vistaview/style.css';

const gallery = vistaView({
  elements: [
    { src: '/images/photo1.jpg', alt: 'Mountain landscape' },
    { src: '/images/photo2.jpg', alt: 'Ocean sunset' },
    { 
      src: '/images/photo3-800.jpg', 
      alt: 'City skyline',
      srcSet: '/images/photo3-800.jpg 800w, /images/photo3-1200.jpg 1200w'
    },
  ],
});

// Open first image
gallery.open(0);
```

### With Extensions

```js
import { vistaView } from 'vistaview';
import { download } from 'vistaview/extensions/download';
import { youtubeVideo } from 'vistaview/extensions/youtube-video';
import 'vistaview/style.css';

const gallery = vistaView({
  elements: '#gallery a',
  extensions: [download(), youtubeVideo()],
  controls: {
    topLeft: ['indexDisplay'],
    topRight: ['download', 'zoomIn', 'zoomOut', 'close'],
    bottomLeft: ['description'],
  },
});
```

### Event Handling

```js
import { vistaView } from 'vistaview';
import 'vistaview/style.css';

const gallery = vistaView({
  elements: '#gallery a',
  onOpen: (vistaView) => {
    console.log('Gallery opened');
  },
  onClose: (vistaView) => {
    console.log('Gallery closed');
  },
  onImageView: (data, vistaView) => {
    console.log(`Viewing image ${data.index.to + 1}`);
  },
});
```
